== Topology
[source,scala]
----
  def dedup = {
    inStream
      .peek{ case (k,v) => logger.debug(s"read from input: ${k}")}
      .leftJoin(duplicatesTblStream)(
        {case(k,v)=> k},
        {case(v,rv) => if(rv==null) v else null}
      )
      .peek{ case (k,v) => logger.debug(s"join result: ${k}")}
      .filter{case(k,v) => v != null}
      .through(duplicatesTable)
      .to(toRemoteTable)
  }
----
